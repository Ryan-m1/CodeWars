class Solution {
    public int searchInsert(int[] nums, int target) {
        int l = 0;
        int r = nums.length - 1;
        while (l <= r) {
            //é˜²æ­¢æº¢å‡ºï¼Œç­‰åŒäº(l+r)/2
            int mid = l + ((r - l) >> 1);
            if (nums[mid] < target) {
                // targetåœ¨å³åŒºé—´
                l = mid + 1;
            } else if (nums[mid] > target) {
                // targetåœ¨å·¦åŒºé—´
                r = mid - 1;

        }
        return r + 1;
    }
}
